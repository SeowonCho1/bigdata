import pandas as pd
import numpy as np

np.random.seed(2026)
n = 60

start = pd.date_range("2024-01-01 08:00", periods=n, freq="3H")
end = start + pd.to_timedelta(np.random.randint(10, 180, n), unit="m")
measure = np.random.randint(10, 50, n).astype(float)
measure[[4, 19, 33]] = [10.5, 21.2, 30.7]

df = pd.DataFrame({
    "id": [f"C{1000+i}" for i in range(n)],
    "name": np.random.choice(["Kim","Lee","Park","Choi","Jung"], n),
    "city": np.random.choice(["Seoul","Busan","Daegu","Incheon"], n),
    "category": np.random.choice(["A","B","C"], n),
    "group": np.random.choice(["G1","G2","G3"], n),
    "score": np.random.normal(70, 12, n).round(1),
    "amount": np.random.randint(100, 1000, n),
    "qty": np.random.randint(1, 10, n),
    "price": np.random.randint(1000, 10000, n),
    "date": pd.date_range("2024-01-01", periods=n, freq="7D"),
    "start_time": start,
    "end_time": end,
    "distance_km": np.random.uniform(1, 120, n).round(2),
    "text": np.random.choice(["AB-001","CD-203","EF-101","GH-404"], n),
    "val1": np.random.randint(1, 100, n),
    "val2": np.random.randint(1, 100, n),
    "val3": np.random.randint(1, 100, n),
    "num_str": np.random.choice(["10","20","30","bad","40"], n),
    "measure": measure
})

df.loc[[3,10,25], "score"] = np.nan
df.loc[[5,12], "city"] = np.nan
df.loc[[8,22], "category"] = np.nan
df.loc[7, "amount"] = 5000
df = pd.concat([df, df.iloc[[0,1]]], ignore_index=True)

df_a = df.iloc[:20].copy()
df_b = df.iloc[20:40].copy()
left = df[["id","city","amount"]].head(15).copy()
right = pd.DataFrame({
    "id": df["id"].head(15),
    "grade": np.random.choice(["Gold","Silver","Bronze"], 15)
})
